update system set srcsys='AssetManager',srcid=systemid 
where (systemid REGEXP '^S[0-9]+$' or systemid REGEXP '^SGER[0-9]+$') and
      cistatus in (3,4,5) and
      srcsys<>'AssetManager';

update asset set srcsys='AssetManager',srcid=name 
where (name REGEXP '^A[0-9]+$' or name REGEXP '^AGER[0-9]+$') and
      cistatus in (3,4,5) and
      srcsys<>'AssetManager';
